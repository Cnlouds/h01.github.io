#### 设置权限
[code:js:]
{
    "permissions": ["management"]
}
[codend]

#### 读取信息
[code:js:]
chrome.management.getAll(function(arr){
    console.log(arr);
}
[codend]

#### 获取权限警告
[code:js:]
chrome.management.getPermissionWarningsById("扩展的ID", function(warning){
    console.log(warning);
}
[codend]

#### 启用禁用扩展
[code:js:]
// true=启用;false=禁用
chrome.management.setEnabled("扩展ID", false, function(){
    console.log(".");
}
[codend]
